
# API 설계
GET - Todo 리스트 전체 조회 - http://localhost:5000/todo
POST - 그룹과 할 일 추가 - http://localhost:5000/todo
PATCH - 할 일 상태 수정 - http://localhost:5000/todo/:id


# 데이터베이스 설계
Database : MongoDB (ODM : mongoose)
collection : todo 
document : items
schema : 
    key(키) : mongoose.Schema.Types.ObjectId
    task(할 일) : String
    group(그룹) : String
    state(상태) : Boolean, default => false
